\begin{itemize}

  \vbox{
  \item [\texttt{BPMs}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de números enteros)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ 60 ]}
    \end{minipage}
    \vskip.2\baselineskip
    Pulsos por minuto.
}
\begin{minted}{yaml}
BPMs: [ 75, 65, 70 ]

\end{minted}
  \vbox{
  \item [\texttt{alturas}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de números enteros)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[1]}
    \end{minipage}
    \vskip.2\baselineskip
    Punteros de intervalo.

Cada miembro es un índice de intervalo en el conjunto: \texttt{registracion}.

}
\begin{minted}{yaml}
alturas: [ 1, 3, 5, 8 ]
\end{minted}
  \vbox{
  \item [\texttt{controles}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de listas de pares)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{None}
    \end{minipage}
    \vskip.2\baselineskip
    Cambios de control. Secuencia de pares, número controlador y valor a asignar.
}
\begin{minted}{yaml}
controles:
  - [ 70 : 80,  71 : 90, 72 : 100 ]
  - [ 33 : 121, 51 : 120 ]
  - [ 10 : 80,  11 : 90, 12 : 100, 13 : 100 ]

\end{minted}
  \vbox{
  \item [\texttt{dinamicas}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de números decimales)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ 1 ]}
    \end{minipage}
    \vskip.2\baselineskip
    Lista ordenada de dinámicas. Factores de 127.
}
\begin{minted}{yaml}
dinamicas: [ 1, .5, .4 ]
\end{minted}
  \vbox{
  \item [\texttt{duraciones}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de números decimales)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ 1 ]}
    \end{minipage}
    \vskip.2\baselineskip
    Largo de articulaciones, lista de factores relativos al de pulso.
}
\begin{minted}{yaml}
duraciones: [ 1, .5, .5, 1, 1 ]
\end{minted}
  \vbox{
  \item [\texttt{letras}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de cadenas de caracteres)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ None ]}
    \end{minipage}
    \vskip.2\baselineskip
    Texto. Versos o anotaciones.
}
\begin{minted}{yaml}
letras: [ 'Hola', 'mundo.' ]
\end{minted}
  \vbox{
  \item [\texttt{programas}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de números enteros)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ None ]}
    \end{minipage}
    \vskip.2\baselineskip
    Lista de números de instrumento MIDI en el banco actual.
}
\begin{minted}{yaml}
programas: [ 25 ] # Guitarra en banco GM
\end{minted}
  \vbox{
  \item [\texttt{tonos}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de enteros)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ 0 ]}
    \end{minipage}
    \vskip.2\baselineskip
    Curva de entonaciones (pitch-bends).
}
\begin{minted}{yaml}
tonos: [ 666, -777, 0 ]
\end{minted}
  \vbox{
  \item [\texttt{voces}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de listas de números enteros)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{None}
    \end{minipage}
    \vskip.2\baselineskip
    Superposición de alturas.
Apilamiento de voces dentro de la \texttt{registracion} fija.

Lista de listas, cada miembro es una voz y cada voz es un lista de enteros que se añade al puntero de intervalo.

}
\begin{minted}{yaml}
voces:
   - [ 8, 6 ] 
   - [ 5 ] 
   - [ 3 ]

\end{minted}

\end{itemize}
