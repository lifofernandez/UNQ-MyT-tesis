\begin{itemize}

  \vbox{
  \item [\texttt{complementos}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(cadena de caracteres)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{None}
    \end{minipage}
    \vskip.2\baselineskip
    Ubicación de modulo con métodos de usuario.
}
\begin{minted}{yaml}
complementos: 'enchufes.py'

\end{minted}
  \vbox{
  \item [\texttt{forma}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de cadenas de caracteres)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ None ]}
    \end{minipage}
    \vskip.2\baselineskip
    Macro estructura de la pista.
Lista de unidades a ser secuenciadas.
Cada elemento corresponde a un miembro de la paleta de unidades.

}
\begin{minted}{yaml}
forma: [
  'espera',
  'estrofa',
  'estribo',
]

\end{minted}
  \vbox{
  \item [\texttt{nombre}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(cadena de caracteres)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{<empty>}
    \end{minipage}
    \vskip.2\baselineskip
    Titulo de la pista
}
\begin{minted}{yaml}
nombre: 'Feliz Cumpleaños'

\end{minted}
  \vbox{
  \item [\texttt{unidades}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(diccionario)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ None ]}
    \end{minipage}
    \vskip.2\baselineskip
    Paleta de unidades disponibles, con respectivas sus propiedades.
}
\begin{minted}{yaml}
unidades: 
  base: &base 
    transportar: 72 
    registracion: [ 
      -12, -10, -8, -7, -5, -3, -1,
        0,   2,  4,  5,  7,  9, 11,
       12,  14, 15, 17, 19, 21, 23,
       24
    ]
    metro: 3/4
    duraciones: [ 1.5 ]
  espera:
    <<: *base
    dinamicas: [ 0 ] 
    duraciones: [ 2 ]
  a: &a 
    <<: *base
    alturas: [ 5, 5, 6, 5 ] 
    duraciones: [ .75, .25, 1, 1 ]
    letras: [ 'Que', 'los', 'cum', 'plas,' ]
  b: &b 
    <<: *base
    alturas: [ 8, 7 ] 
    duraciones: [ 1, 2 ]
    letras: [ 'fe', 'liz.' ]
  b`: 
    <<: *b
    transponer: 1
  a`: 
    <<: *a
    alturas: [ 5, 5, 12, 10 ] 
  a``: 
    <<: *a
    alturas: [ 8, 8, 7, 6 ] 
    letras: [ 'que', 'los', 'cum', 'plas,' ]
  a```: 
    <<: *a
    alturas: [ 11, 11, 10, 8 ] 
    letras: [ 'que', 'los', 'cum', 'plas,' ]
  A: 
    forma: [ 'a', 'b' ] 
  A`: 
    forma: [ 'a' , 'b`' ]
  B: 
    forma: [ 'a`', 'a``' ] 
  A``: 
    forma: [ 'a```', 'b`' ] 
  estrofa: 
    forma: [ 'A', 'A`' ]
  estribo: 
    forma: [ 'B', 'A``' ]

\end{minted}

\end{itemize}
