\begin{itemize}

  \vbox{
  \item [\texttt{nombre}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(cadena de caracteres)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{<empty>}
    \end{minipage}
    \vskip.2\baselineskip
    \small{ Titulo de la pista. }
}
\begin{minted}{yaml}
nombre: 'Feliz Cumpleaños'

\end{minted}
  \vbox{
  \item [\texttt{unidades}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(diccionario)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ None ]}
    \end{minipage}
    \vskip.2\baselineskip
    \small{ Paleta de unidades disponibles (con sus propiedades respectivas) para ser invocadas en \texttt{forma}. }
}
\begin{minted}{yaml}
unidades: 
  base: &base 
    transportar: 72 
    registracion: [ 0, 2, 4, 5, 7, 9, 11, 12 ]
    metro: 3/4
  a: &a 
    <<: *base
    alturas: [ 5, 5, 6, 5 ] 
    duraciones: [ .75, .25, 1, 1 ]
  b: &b 
    <<: *base
    alturas: [ 8, 7 ] 
    duraciones: [ 1, 2 ]
  motivoA: 
    forma: [ 'a', 'b' ] 
  motivoB: 
    forma: [ 'b', 'b' ] 

\end{minted}
  \vbox{
  \item [\texttt{forma}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(lista de cadenas de caracteres)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{[ None ]}
    \end{minipage}
    \vskip.2\baselineskip
    \small{ Macro estructura de la pista.
Lista de unidades a ser secuenciadas,
cada elemento corresponde a un miembro de la paleta de \texttt{unidades}.
 }
}
\begin{minted}{yaml}
forma: [ 'motivoA', 'motivoB' ]

\end{minted}
  \vbox{
  \item [\texttt{complementos}] 
    \begin{minipage}[t]{6cm}\flushleft
	    \textsf{(cadena de caracteres)}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{6cm}\flushright
      default: \texttt{None}
    \end{minipage}
    \vskip.2\baselineskip
    \small{  \hypertarget{complementos}{}Ubicación de módulo con métodos de usuario. }
}
\begin{minted}{yaml}
complementos: 'enchufes.py'

\end{minted}

\end{itemize}
